#include "ReverseQueue.h"
#include <stack>
#include <iostream>
using namespace std;

void ReverseFirstK(queue<int>& q, int k)
{
    if (k <= 0 || k > q.size())
    {
        cout << "Invalid value of K!" << endl;
        return;
    }

    stack<int> s;

    for (int i = 0; i < k; i++)
    {
        s.push(q.front());
        q.pop();
    }

    while (!s.empty())
    {
        q.push(s.top());
        s.pop();
    }

    int remaining = q.size() - k;

    for (int i = 0; i < remaining; i++)
    {
        q.push(q.front());
        q.pop();
    }
}
